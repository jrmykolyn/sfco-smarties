// No options.

var ref = new Smarties();

// Options object with `delay` key/value pair.
// Value represents the number of millisecond between instantiation and first insertion.

var ref = new Smarties( {
    delay: 10000,
} );

// Options object with `intervalLength` key/value pair.
// Value represents the number of milliseconds between insertions.

var ref = new Smarties( {
    intervalLength: 250,
} );

// Options object with `backgroundColors` array.
// NOTE: Providing 1x value overrides the default background color of the Smarties elements.

var ref = new Smarties( {
    backgroundColors: [
        'darkslateblue',
    ],
} );

// Options object with `backgroundColors` array.
// NOTE: Providing 2 or more items allows Smarties to randomly select from the available background colors at insertion time.

var ref = new Smarties( {
    backgroundColors: [
        'darkslateblue',
        '#ff0022',
        'rgb( 0, 200, 50 )',
        'rgba( 50, 100, 200, 0.5 )',
    ],
} );

// Options object with `sizes` array.
// NOTE: Providing 1x value overrides the default size of the Smarties elements.

var ref = new Smarties( {
    sizes: [ '200px' ],
} );

// Options object with `sizes` array.
// NOTE: Providing 2 or more items allows Smarties to randomly select from the available sizes at insertion time.

var ref = new Smarties( {
    sizes: [
        '75px',
        '100px',
        '12.5rem',
        '2em'
    ],
} );

// Options object with `images` array.
// NOTE: Providing 1x value replaces the default background color with the specified image.

var ref = new Smarties( {
    images: [ './assets/images/segal_1.jpg' ],
} );

// Options object with `images` array.
// NOTE: Providing 2 or more items allows Smarties to randomly select from the available images at insertion time.

var ref = new Smarties( {
    images: [
        './assets/images/robo_1.png',
        './assets/images/robo_2.png',
        './assets/images/robo_3.png',
        './assets/images/robo_4.png',
        './assets/images/robo_5.png',
    ],
} );

// Options object with `images`, `sizes`, `backgroundColors`, and `forceBackgroundColor`.

var ref = new Smarties( {
    images: [
        './assets/images/robo_1.png',
        './assets/images/robo_2.png',
        './assets/images/robo_3.png',
        './assets/images/robo_4.png',
        './assets/images/robo_5.png',
    ],
    sizes: [
        '60px',
        '100px',
        '140px',
    ],
    backgroundColors: [
        'darkslateblue',
        '#ff0022',
        'rgb( 0, 200, 50 )',
        'rgba( 50, 100, 200, 0.5 )',
    ],
    forceBackgroundColor: true,
} );

// Options object with `callbacks`. The `preInsert` callback can be used to maniplate each node prior to insertion.

var count = 0;
var ref = new Smarties( {
    backgroundColors: [
        'red',
        'green',
        'blue',
    ],
    callbacks: {
        preInsert: function( nodeElem ) {
            nodeElem.classList.add( 'count-elem' ); // Add a custom class to each node.
            nodeElem.innerHTML = ++count; // Insert custom content into each node.
        },
    },
} );

// Options object with `callbacks`. The `postInsert` callback can be used to maniplate each node after insertion.

var count = 0;
var ref = new Smarties( {
    backgroundColors: [
        'red',
        'green',
        'blue',
    ],
    callbacks: {
        preInsert: function( nodeElem ) {
            nodeElem.classList.add( 'grow-elem--pre' );
        },
        postInsert: function( nodeElem ) {
            // Add a custom class 500ms after insertion.
            setTimeout( function() {
                nodeElem.classList.add( 'grow-elem--post' );
            }, 500 );
        },
    },
} );

// Options object with `callbacks`. The `preStop` callback can be used to manipulate the document before the `stop()` method is called.

var ref = new Smarties( {
    callbacks: {
        preStop: function( nodes ) {
            alert( 'Inside `preStop` callback!' );
        },
    },
} );

setTimeout( function() {
    ref.stop();
}, 2000 );

// Options object with `callbacks`. The `preStop` callback can be used to manipulate the document after the `stop()` method is called.

var ref = new Smarties( {
    callbacks: {
        postStop: function( nodes ) {
            alert( 'Inside `postStop` callback!' );
        },
    },
} );

setTimeout( function() {
    ref.stop();
}, 2000 );
